GRANT CONNECT TO HR;
GRANT CONNECT, resource, dba TO HR;

GRANT CREATE SESSION  TO HR; -- deu erro

GRANT unlimited tablespace TO HR;
GRANT SELECT , INSERT, UPDATE TO HR;
