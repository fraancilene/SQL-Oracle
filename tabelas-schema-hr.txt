-- tabelas do schema HR, tem que finalizar com as chaves primarias e estrangeiras

CREATE TABLE tb_location(
	location_id number(4) NOT NULL,
	street_Address varchar2(40),
	postal_code varchar2(12),
	city varchar2(30),
	state_province varchar2(10),
	country_id number(4)
);

CREATE TABLE tb_countries(
	country_id char(2) NOT NULL,
	country_name varchar2(40),
	region_id number
);

CREATE TABLE tb_regions(
	region_id number NOT NULL,
	region_name varchar(2)
);

CREATE TABLE tb_departaments(
	department_id number(4) NOT NULL,
	department_name varchar2(30) NOT NULL,
	manager_id number(6),
	location_id number(4)
);

CREATE TABLE tb_employees(
	employee_id number(6) NOT NULL,
	first_name varchar2 (20),
	last_name varchar2 (25) NO NULL,
	email varchar2 (25) NOT NULL,
	phone_number varchar2 (20),
	hire_date date NOT NULL,
	job_id varchar2(10) NOT NULL,
	salary number (8,2),
	commission_pct number (2,2),
	manager_id number(6),
	department_id number(4)
	
);

CREATE TABLE tb_jobs(
	job_id varchar2(10) NOT NULL,
	job_title varchar2(35), NOT NULL,
	min_salary number(6),
	max_salary number(6)
);

CREATE TABLE tb_job_history(
	employee_id number(6) NOT NULL,
	start_Date date NOT NULL,
	end_date date NOT NULL,
	job_id varchar2(10) NOT NULL,
	department_id number(4)
);
